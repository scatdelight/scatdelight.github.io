```c++
#include<iostream>
#include<string>
#include<vector>
#include<sstream>

using namespace std;

int main()
{
    string str="hello world";
    istringstream ss(str);
    string stringBuffer;
    vector<string> v;

    while (getline(ss, stringBuffer, ' ')){
        v.push_back(stringBuffer);
        cout<<stringBuffer<<" ";
    }
    
    return 0;
}
```
